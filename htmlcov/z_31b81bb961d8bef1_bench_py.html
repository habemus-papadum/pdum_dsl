1"""Measurement as a satellite: adaptive microbenchmarks, phase timers, timelines. 

2 

3Three tools, none of which the kernel knows about (zero kernel edits): 

4 

5- ``benchmark(fn)`` — BenchmarkTools.jl-style adaptive sampling. One warmup 

6 call; then evals-per-sample is TUNED upward until a sample takes long 

7 enough to swamp timer resolution; then samples accumulate until the time 

8 budget. The **minimum** is the headline estimator (noise is strictly 

9 additive on a quiet machine); median/mean report the distribution. 

10 A naive ``timeit``-one-loop mean overstates fast paths — ch11b shows by 

11 how much. 

12- ``instrument(target, ...)`` — phase decomposition of the dispatch hot 

13 path, by *temporarily wrapping the FastRecord's seams* (``extract`` and 

14 ``launch`` are plain dataclass fields): key+probe, extract, pack, launch 

15 fall out as timestamp differences. Restores the record afterwards. 

16- ``Timeline`` — a list of labelled spans, rendered by the viz satellite as 

17 a static-HTML bar chart (hover for µs; one lane per phase family). 

18 

19GPU decomposition rides the demo runtime's ``timed_call`` (WebGPU 

20``timestamp-query``: begin/end-of-pass timestamps in nanoseconds), which 

21splits ``launch`` into encode / GPU execution / readback — the instrument 

22that answers ch10's "where did the 2 ms go". 

23"""

34@dataclass 

35class Trial: 

36 """One benchmark's evidence: per-sample seconds (each averaged over 

37 ``evals`` inner calls), plus the tuning that produced them.""" 

38 

39 samples: list # seconds per single eval (already divided by evals) 

40 evals: int 

41 warmup_s: float 

42 

43 @property 

44 def minimum(self) -> float: 

45 return min(self.samples) 

46 

47 @property 

48 def median(self) -> float: 

49 return statistics.median(self.samples) 

50 

51 @property 

52 def mean(self) -> float: 

53 return statistics.fmean(self.samples) 

54 

55 def __repr__(self) -> str: 

56 u = 1e6 

57 return ( 

58 f"Trial(min {self.minimum * u:.2f} µs · median {self.median * u:.2f} µs · " 

59 f"mean {self.mean * u:.2f} µs · {len(self.samples)} samples × {self.evals} evals)" 

60 )

63def benchmark( 

64 fn, *, budget_s: float = 0.25, min_sample_s: float = 50e-6, max_samples: int = 10_000, timer=_pc 

65) -> Trial: 

66 """Adaptive microbenchmark. ``min_sample_s`` is the resolution floor: the 

67 per-sample duration is pushed above it by raising evals-per-sample, so 

68 quantization noise cannot masquerade as speed.""" 

69 t0 = timer() 

70 fn() # warmup: caches hot, imports done, JIT warmed 

71 warmup = timer() - t0 

72 evals = 1 

73 while True: # tune: double evals until one sample clears the floor 

74 t0 = timer() 

75 for _ in range(evals): 

76 fn() 

77 if (timer() - t0) >= min_sample_s or evals >= 1 << 20: 

78 break 

79 evals <<= 1 

80 samples: list = [] 

81 spent, deadline = 0.0, budget_s 

82 while spent < deadline and len(samples) < max_samples: 

83 t0 = timer() 

84 for _ in range(evals): 

85 fn() 

86 dt = timer() - t0 

87 samples.append(dt / evals) 

88 spent += dt 

89 return Trial(samples, evals, warmup)

93class Timeline: 

94 """Labelled spans for the viz timeline widget: (label, start_s, dur_s, lane)."""

107def _warm_record(registry, target, args, out): 

108 """Warm, then locate, the FastRecord that dispatch just used.""" 

109 registry.dispatch(target, args, out) 

110 arg_fp = tuple(registry.table.fingerprint(a) for a in args) 

111 key = registry.specializations.key_for(target, arg_fp, registry.backend_for(target.kind).fp) 

112 record = registry.specializations.probe(key) 

113 if record is None: 

114 raise RuntimeError("no warm record to instrument (guards drifting?)") 

115 return record

118def instrument(target, *args, out=None, registry=None, frames: int = 50) -> dict: 

119 """Decompose the dispatch hot path into phases, averaged over ``frames``. 

120 

121 Wraps the warm FastRecord's ``extract`` and ``launch`` fields with 

122 timestamping shims (restored afterwards). Phase math: ``key+probe`` is 

123 what happens before extract begins; ``pack`` is the gap between extract 

124 returning and launch starting. Returns seconds per phase plus 'total'. 

125 """ 

126 registry = registry or DEFAULT 

127 record = _warm_record(registry, target, args, out) 

128 marks: dict = {} 

129 orig_extract, orig_launch = record.extract, record.launch 

130 

131 def timed_extract(captures, call_args): 

132 marks["extract_start"] = _pc() 

133 result = orig_extract(captures, call_args) 

134 marks["extract_end"] = _pc() 

135 return result 

136 

137 def timed_launch(staging, leaves): 

138 marks["launch_start"] = _pc() 

139 result = orig_launch(staging, leaves) 

140 marks["launch_end"] = _pc() 

141 return result 

142 

143 record.extract, record.launch = timed_extract, timed_launch 

144 sums = {"key+probe": 0.0, "extract": 0.0, "pack": 0.0, "launch": 0.0, "total": 0.0} 

145 try: 

146 for _ in range(frames): 

147 marks.clear() # a mid-loop rebuild would leave last frame's marks — fail loud instead 

148 t0 = _pc() 

149 registry.dispatch(target, args, out) 

150 t1 = _pc() 

151 if "launch_end" not in marks: # dispatch served a fresh, unwrapped record 

152 raise RuntimeError("record was rebuilt mid-instrument (guard drift?)") 

153 sums["key+probe"] += marks["extract_start"] - t0 

154 sums["extract"] += marks["extract_end"] - marks["extract_start"] 

155 sums["pack"] += marks["launch_start"] - marks["extract_end"] 

156 sums["launch"] += marks["launch_end"] - marks["launch_start"] 

157 sums["total"] += t1 - t0 

158 finally: 

159 record.extract, record.launch = orig_extract, orig_launch 

160 return {k: v / frames for k, v in sums.items()}

163def phase_timeline(phases: dict, title: str = "dispatch") -> Timeline: 

164 """Order the instrument() phases into a single-lane host timeline.""" 

165 tl = Timeline(title=title) 

166 cursor = 0.0 

167 for name in ("key+probe", "extract", "pack", "launch"): 

168 tl.add(name, cursor, phases[name], "host") 

169 cursor += phases[name] 

170 return tl

173def gpu_timeline(target, *, out, registry=None, frames: int = 20) -> Timeline | None: 

174 """Split `launch` further for the demo WGSL runtime: encode+submit / GPU 

175 execution (timestamp-query, ns) / readback. None when the artifact has no 

176 ``timed_call`` (non-GPU targets) or the adapter lacks the feature. 

177 

178 The frames are driven by ``registry.dispatch`` itself, with ``launch`` 

179 shimmed to the artifact's ``timed_call`` — the SAME seam-wrap technique as 

180 ``instrument``, so the timed frame is exactly the frame the hot path runs 

181 (no hand-replayed extract/pack that could drift from dispatch).""" 

182 registry = registry or DEFAULT 

183 record = _warm_record(registry, target, (), out) 

184 timed = getattr(record.artifact, "timed_call", None) 

185 if timed is None: 

186 return None 

187 box: dict = {} 

188 orig_launch = record.launch 

189 record.launch = lambda staging, leaves: box.__setitem__("parts", timed(staging, leaves)) 

190 agg: dict = {} 

191 try: 

192 for _ in range(frames): 

193 box.clear() 

194 registry.dispatch(target, (), out) 

195 if "parts" not in box: # dispatch served a fresh, unwrapped record 

196 raise RuntimeError("record was rebuilt mid-instrument (guard drift?)") 

197 if box["parts"] is None: 

198 return None # adapter lacks timestamp-query 

199 for k, v in box["parts"].items(): 

200 agg[k] = agg.get(k, 0.0) + v / frames 

201 finally: 

202 record.launch = orig_launch 

203 tl = Timeline(title="one GPU frame") 

204 cursor = 0.0 

205 for name in ("encode+submit", "gpu", "readback"): 

206 tl.add(name, cursor, agg[name], "gpu" if name == "gpu" else "host") 

207 cursor += agg[name] 

208 return tl
