1"""The Registry (surface E, v1) and the hot path: the thesis, mechanically. 

2 

3One explicit object owns what the staged seeds have been accumulating: 

4the kind table, the lowering rule packs, the Derived build rules, the 

5backends, and the two cache tiers. Satellites register INTO it at import 

6(``pdum.dsl.stdlib`` brings the base language, ``pdum.dsl.backends.python`` 

7brings the first backend); the kernel never imports a satellite. 

8 

9``dispatch`` is the whole runtime: 

10 

11- **hit** — fingerprint the args (types, never values), build the key, 

12 probe tier 1, check guards, then run the precompiled ``FastRecord``: 

13 compiled extract → generic pack into reused staging → ``launch``. 

14 No parse, no typeof, no IR, no plan — those all happened once, at miss. 

15- **miss** — §4.3 in order: typeof → lower (coherence inside) → NORMALIZE → 

16 legalize_params(plan) → tier-2 probe on the *content key* (identical IR 

17 never renders twice, even from different templates) → render+compile → 

18 plan/extractor/guards → ``FastRecord`` installed under a per-key future. 

19 

20Guards are the drift police: for every captured cell, ``(cell, 

21"cell_contents", captured_value)`` — an *identity* triple. A capture rebound 

22after decoration is caught at the next call (guard miss, loud counter, 

23recompile against the same frozen env) rather than silently serving bytes 

24from a world that no longer exists. Edited-and-rerun templates retire their 

25predecessors via the cache's code map. 

26 

27``Backend`` carries name/render/compile/fp plus the step-9 columns 

28(``plan``, ``param_types``, ``make_launcher``). Still deferred from §2.10: 

29``code_for_op``-gated decompositions and a real ``params_key`` column — 

30type_map was absorbed into the wgsl renderer's tables, and the fragment 

31target format rides ``fp`` (coarse but sound) until multiple formats exist. 

32 

33Book: ``docs/book/ch09-end-to-end-on-cpu.ipynb``. 

34"""

52@dataclass(frozen=True) 

53class Backend: 

54 """A capability record (§2.10; the step-9 columns arrive with the second 

55 backend). The three optional callables let a backend own its layout, its 

56 parameter convention, and its launcher without the kernel knowing any 

57 target's shape — the python backend takes every default.""" 

58 

59 name: str 

60 render: Callable # (region, plan, backend) -> source text; backend = THIS record (spelling table) 

61 compile: Callable # (source, name) -> artifact 

62 fp: tuple # enters every specialization key (a backend/version change is a new world) 

63 plan: Callable | None = None # (env_types, arg_types, table) -> PackPlan; None = dense staging 

64 param_types: Callable | None = None # (target) -> tuple | None; a family that DERIVES params 

65 # (compute: params ARE thread coords) returns their types; None = typeof the call args 

66 make_launcher: Callable | None = None # (artifact, plan) -> launch(staging, leaves); None = artifact 

67 code_for_op: dict = field(default_factory=dict) # op -> spelling template; keys() = capability set

70class NoBackend(RuntimeError): 

71 """Dispatch reached a registry with no backend — ``import pdum.dsl`` for 

72 batteries, or ``register_backend`` your own.""" 

73 

74 

75# Out tags the ``out=`` payload on the leaves channel: launchers peel it by 

76# TYPE, never tail position — buffer leaves (ch12) cannot collide with it. 

77Out = namedtuple("Out", "value")

80def _guards(target) -> tuple: 

81 if isinstance(target, Handle): 

82 cells = dict(zip(target.freevars, target.pyfunc.__closure__ or ())) 

83 own = tuple((cells[name], "cell_contents", expected) for name, expected in target.env.items()) 

84 # Recurse into captured kernels: an INNER handle's cell drifting is the 

85 # same staleness as an outer one (review-caught hole — inlined callees 

86 # are baked into the artifact, so their drift must guard the entry). 

87 return own + tuple(g for v in target.captures if isinstance(v, Handle) for g in _guards(v)) 

88 return tuple(g for h in target.captures for g in _guards(h)) # a Pipeline guards its stages

91class Registry: 

92 """v1: enough for backends + rule packs + the dispatch loop. Layering 

93 (``extend()``), overloads, and Dialect bundles complete it at step 10.""" 

94 

95 def __init__(self, table: KindTable = BUILTINS): 

96 self.table = table 

97 self.lower_rules: dict = {} 

98 self.derived: dict = {} 

99 self.ops: dict = {} # surface A: dialect OpDefs, merged into every build 

100 self.overloads: dict = {} # surface B: call-name -> impl (satellites interpret) 

101 self.decompositions: list = [] # (op_name, rule): applied when the backend lacks the op 

102 self.backends: dict[str, Backend] = {} 

103 self.routes: dict[str, str] = {} # kind -> backend name; absent kinds use the default 

104 self.default_backend: str | None = None 

105 self.specializations = SpecializationCache() 

106 self.artifacts = ArtifactCache() 

107 

108 def register_backend(self, backend: Backend, *, default: bool = False, kinds: tuple = ()) -> None: 

109 self.backends[backend.name] = backend 

110 self.routes.update(dict.fromkeys(kinds, backend.name)) # roles ship WITH their backends 

111 if default or (self.default_backend is None and not kinds): 

112 # A kind-scoped backend never CLAIMS the default slot: a registry 

113 # holding only wgsl must refuse a "device" kernel loudly, not 

114 # silently compile it through the compute backend (review-caught). 

115 self.default_backend = backend.name 

116 

117 def backend_for(self, kind: str) -> Backend: 

118 if (name := self.routes.get(kind, self.default_backend)) is None: 

119 raise NoBackend("no backend registered; `import pdum.dsl` wires the batteries") 

120 return self.backends[name] 

121 

122 def dispatch(self, target, args: tuple, out=None): 

123 """THE hot path. `target` is anything FnType-shaped: Handle or Pipeline. 

124 `out` is launcher data (destinations / launch domain) — it rides the 

125 leaves channel and never touches any key (070 §3: grid strips).""" 

126 spec, table = self.specializations, self.table 

127 backend = self.backend_for(target.kind) 

128 key = (target.fp, tuple(table.fingerprint(a) for a in args), backend.fp, spec.generation) 

129 record = spec.probe(key) or spec.get_or_compile(key, lambda: self._build(target, args, backend)) 

130 leaves = pack_into(record.plan, record.staging, record.extract(target.captures, args)) 

131 return record.launch(record.staging, leaves if out is None else (*leaves, Out(out))) 

132 

133 def _build(self, target, args: tuple, backend: Backend) -> FastRecord: 

134 """The miss path (§4.3), once per (types, backend, generation).""" 

135 derived_params = backend.param_types(target) if backend.param_types else None 

136 if derived_params is not None and args: # a positional arg would be silently dead: refuse 

137 raise VerifyError(f"{backend.name} derives the params; pass the launch domain via out=") 

138 arg_types = derived_params if derived_params is not None else tuple(self.table.typeof(a) for a in args) 

139 ops = {**CORE_OPS, **ABI_OPS, **self.ops} 

140 rules = {**self.lower_rules, "__registry__": self} # rule packs' context door (surface B lives there) 

141 region = lower_handle(target, rules, ops, arg_types=arg_types, derived=self.derived) 

142 plan = (backend.plan or plan_from_types)(target.env_types, arg_types, self.table) 

143 gated = [r for op, r in self.decompositions if op not in backend.code_for_op] 

144 if gated: # shared decompositions run only where the target lacks the native op (§2.10) 

145 region = rewrite(region, gated, ops, name="decompose") 

146 region = run_stage(region, NORMALIZE_ENV, ops) 

147 dialects = frozenset(op.split(".", 1)[0] for op in self.ops) 

148 region = run_stage(region, legalize_params(plan, dialects), ops) 

149 artifact = self.artifacts.get_or_compile( # content-addressed: identical IR compiles once 

150 (region.key, backend.fp), # fp, not name: a backend VERSION change is a new artifact world 

151 lambda: backend.compile(backend.render(region, plan, backend)), 

152 ) 

153 extract = build_extractor(target.env_types, arg_types, plan, self.table) 

154 return FastRecord( 

155 artifact=artifact, 

156 guards=_guards(target), 

157 extract=extract, 

158 plan=plan, 

159 staging=bytearray(plan.staging_size), 

160 launch=backend.make_launcher(artifact, plan) if backend.make_launcher else artifact, 

161 ) 

162 

163 def extend(self) -> Registry: 

164 """Surface-E layering (stdlib -> user -> session): a child registry 

165 with copied registrations, an extended kind table, FRESH caches.""" 

166 child = Registry(self.table.extend()) 

167 for a in ("lower_rules", "derived", "routes", "ops", "overloads"): 

168 getattr(child, a).update(getattr(self, a)) 

169 # Backend records are shared-immutable EXCEPT code_for_op: copy it, or a 

170 # child's spell() would mutate the parent's table (review-caught leak). 

171 child.backends = {k: replace(b, code_for_op=dict(b.code_for_op)) for k, b in self.backends.items()} 

172 child.decompositions, child.default_backend = list(self.decompositions), self.default_backend 

173 return child

175 def load_entry_points(self, group: str = "pdum.dsl.backends", entries=None) -> list: 

176 """Third-party discovery (080 §4): each entry point names an 

177 ``install(registry)``; explicit, lazy, no import-order dependence.""" 

178 from importlib.metadata import entry_points 

179 

180 eps = list(entries) if entries is not None else list(entry_points(group=group)) 

181 for ep in eps: 

182 ep.load()(self) 

183 return [ep.name for ep in eps]
