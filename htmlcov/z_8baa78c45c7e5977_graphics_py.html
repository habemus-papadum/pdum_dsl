1"""Demo vocabulary: a toy ``Color`` record + 2D vector helpers. 

2 

3This is a DEMO on purpose (090's stdlib-minimalism policy). Real color 

4modeling — spaces, RGB vs Lab vs OkLab, gamma — is a domain library's 

5ground, and the five surfaces make such a library an ordinary pip install 

6(an ``install(registry)`` plus an entry point). The stdlib must not squat 

7on that ground; it proves extension is cheap by staying small. Consuming 

8demo vocabulary is therefore one explicit import:: 

9 

10 from pdum.dsl.demo import graphics # wires Color + helpers into DEFAULT 

11 

12Everything here is portable by construction: the helpers are ``@overload`` 

13DSL bodies (inlined per call site, zero per-target spellings) and ``Color`` 

14is an ``@record`` dataclass (fields → uniform slots, methods inline). 

15 

16The helpers are MODULE-LEVEL on purpose: cross-references (``length2`` → 

17``dot2``/``sqrt``, ``lerp2`` → ``mix``) are globals — invisible to capture, 

18resolved through the overload table at lower time — even when the referent 

19lives in another package, as ``mix`` (stdlib battery) does here. 

20""" 

21 

22from __future__ import annotations 

23 

24from dataclasses import dataclass 

25 

26from ..kernel.registry import DEFAULT 

27from ..stdlib.surfaces import overload, record

30def dot2(a, b): 

31 return a[0] * b[0] + a[1] * b[1] 

32 

33 

34def length2(v): 

35 return sqrt(dot2(v, v)) 

36 

37 

38def lerp2(a, b, t): 

39 return (mix(a[0], b[0], t), mix(a[1], b[1], t)) 

40

42# GL derivative vocabulary (110 §3): one-line sugar over the analytic `D`. 

43# GLSL's dFdx is a finite difference across the pixel QUAD; ours is exact — 

44# and compute shaders have no quads, so analytic is the only derivative there. 

45def ddx(v): 

46 return D(v)[0] 

47 

48 

49def ddy(v): 

50 return D(v)[1] 

51 

52 

53def fwidth(v): 

54 d = D(v) 

55 return abs(d[0]) + abs(d[1]) 

56

58@dataclass(frozen=True) 

59class Color: 

60 r: float 

61 g: float 

62 b: float 

63 

64 def luminance(self): 

65 return 0.2126 * self.r + 0.7152 * self.g + 0.0722 * self.b 

66 

67 def scaled(self, k): 

68 return (self.r * k, self.g * k, self.b * k) 

69 

70 

71def install(registry) -> None: 

72 for fn in (dot2, length2, lerp2, ddx, ddy, fwidth): 

73 overload(registry, fn.__name__)(fn) 

74 record(registry, Color) 

75 

76 

77install(DEFAULT)
