1"""The base-language rule pack: the accepted Python subset, as data. 

2 

3This is a SATELLITE. Each entry maps an ``ast`` node type to a lowering rule; 

4widening the language means adding an entry here (or in another dialect's 

5pack) — never editing the kernel driver. The pack's policies are the base 

6dialect's, chosen deliberately: 

7 

8- **Strict, like the core** (settled at the ch05 walkthrough): no automatic 

9 cast insertion. ``i64 + f64`` is a loud, loc-bearing error; write 

10 ``float(i)``. Literals are honest (``1`` is i64, ``1.0`` is f64) — in a 

11 float expression, write float literals. A friendlier dialect may choose to 

12 auto-insert casts in ITS pack; this one does not. 

13- ``float(x)`` / ``int(x)`` / ``bool(x)`` are the base cast vocabulary, 

14 lowering to explicit ``core.cast`` (visible in IR, inside the content key). 

15- Calls resolve: cast builtins -> captured-Handle inlining -> loud 

16 MissingRule (the overload surface arrives with the Registry, step 8). 

17""" 

18 

19from __future__ import annotations 

20 

21import ast 

22 

23from ..kernel.lower import MissingRule, fmt 

24from ..kernel.types import boolean, f64, i64 

25from ..kernel.valuekind import typeof 

26 

27_BINOPS = {ast.Add: "core.add", ast.Sub: "core.sub", ast.Mult: "core.mul", ast.Div: "core.div", 

28 ast.Mod: "core.mod", ast.Pow: "core.pow"} 

29_PREDS = {ast.Lt: "lt", ast.Gt: "gt", ast.LtE: "le", ast.GtE: "ge", ast.Eq: "eq", ast.NotEq: "ne"} 

30_CASTS = {"float": f64, "int": i64, "bool": boolean} 

31 

32 

33def _expr_stmt(ctx, node): 

34 if isinstance(node.value, ast.Constant): 

35 return None # docstrings and stray constants vanish 

36 raise MissingRule(f"expression statements have no effect here [{fmt(ctx.loc(node))}]") 

37 

38 

39def _assign(ctx, node): 

40 if len(node.targets) != 1 or not isinstance(node.targets[0], ast.Name): 

41 raise MissingRule(f"only single-name assignment is in the base pack [{fmt(ctx.loc(node))}]") 

42 ctx.locals[node.targets[0].id] = ctx.lower(node.value) 

43 return None 

44 

45 

46def _return(ctx, node): 

47 if node.value is None: 

48 raise MissingRule(f"bare return is not a value [{fmt(ctx.loc(node))}]") 

49 return ctx.lower(node.value) 

50 

51 

52def _constant(ctx, node): 

53 if isinstance(node.value, (bool, int, float)): 

54 return ctx.emit("core.const", node=node, type=typeof(node.value), value=node.value) 

55 raise MissingRule(f"literal {node.value!r} has no base-pack lowering [{fmt(ctx.loc(node))}]") 

56 

57 

58def _name(ctx, node): 

59 resolved = ctx.resolve(node.id, node) 

60 if isinstance(resolved, tuple) and resolved[0] == "callee": 

61 raise MissingRule(f"a captured kernel is callable, not a value (yet) [{fmt(ctx.loc(node))}]") 

62 return resolved 

63 

64 

65def _binop(ctx, node): 

66 op = _BINOPS.get(type(node.op)) 

67 if op is None: 

68 raise MissingRule(f"operator {type(node.op).__name__} is not in the base pack [{fmt(ctx.loc(node))}]") 

69 return ctx.emit(op, ctx.lower(node.left), ctx.lower(node.right), node=node) 

70 

71 

72def _unary(ctx, node): 

73 if isinstance(node.op, ast.USub): 

74 return ctx.emit("core.neg", ctx.lower(node.operand), node=node) 

75 raise MissingRule(f"unary {type(node.op).__name__} is not in the base pack [{fmt(ctx.loc(node))}]") 

76 

77 

78def _compare(ctx, node): 

79 if len(node.ops) != 1: 

80 raise MissingRule(f"chained comparison is not in the base pack [{fmt(ctx.loc(node))}]") 

81 pred = _PREDS.get(type(node.ops[0])) 

82 if pred is None: 

83 raise MissingRule(f"comparison {type(node.ops[0]).__name__} is not in the base pack [{fmt(ctx.loc(node))}]") 

84 return ctx.emit("core.cmp", ctx.lower(node.left), ctx.lower(node.comparators[0]), node=node, pred=pred) 

85 

86 

87def _ifexp(ctx, node): 

88 from ..kernel.ir import Region 

89 

90 cond = ctx.lower(node.test) 

91 then = Region(body=(ctx.emit("core.yield", ctx.lower(node.body), node=node.body),)) 

92 other = Region(body=(ctx.emit("core.yield", ctx.lower(node.orelse), node=node.orelse),)) 

93 return ctx.emit("core.if", cond, regions=(then, other), node=node) 

94 

95 

96def _call(ctx, node): 

97 if not isinstance(node.func, ast.Name) or node.keywords: 

98 raise MissingRule(f"only simple positional calls are in the base pack [{fmt(ctx.loc(node))}]") 

99 args = tuple(ctx.lower(a) for a in node.args) 

100 name = node.func.id 

101 if name in _CASTS and len(args) == 1: 

102 return ctx.emit("core.cast", args[0], node=node, to=_CASTS[name]) 

103 from ..kernel.lower import NameFateError 

104 

105 resolved = None 

106 try: 

107 resolved = ctx.resolve(name, node) 

108 except NameFateError: 

109 pass # unknown name: fall through to the richer cannot-call message 

110 if isinstance(resolved, tuple) and resolved[0] == "callee": 

111 return ctx.inline(resolved[1], resolved[2], args, node) 

112 raise MissingRule( 

113 f"cannot call {name!r}: not a cast, not a captured kernel; overloads arrive with the " 

114 f"Registry (surface B) [{fmt(ctx.loc(node))}]" 

115 ) 

116 

117 

118LOWER_RULES = { 

119 ast.Expr: _expr_stmt, 

120 ast.Assign: _assign, 

121 ast.Return: _return, 

122 ast.Constant: _constant, 

123 ast.Name: _name, 

124 ast.BinOp: _binop, 

125 ast.UnaryOp: _unary, 

126 ast.Compare: _compare, 

127 ast.IfExp: _ifexp, 

128 ast.Call: _call, 

129}
