1"""The Python backend: the reference target, and the proof the seam is real. 

2 

3Renders a legalized Region to *readable* Python source (ch09 prints it) and 

4executes it with the exact ABI every real target uses: ``kernel(staging, 

5leaves)``, where every input — captures AND scalar arguments — arrives as 

6bytes in the staging buffer, unpacked at the offsets the plan chose. That is 

7deliberately NOT the fastest way to call Python from Python; it is the same 

8calling convention as a uniform buffer, so the marshaling story is proven 

9end-to-end on CPU before a GPU ever sees it. 

10 

11Honest deviations of the reference target, stated once: 

12 

13- ``core.if`` renders as a real ``if``/``else`` statement — branches are 

14 LAZY, so the guard-then-divide idiom (``1.0/x if x > 0.0 else 0.0``) is 

15 safe (the step-8 review caught the eager first draft crashing on exactly 

16 the guarded input). Each node is emitted in the deepest region that 

17 dominates all its uses: branch-exclusive work stays inside its branch; 

18 anything shared hoists to the join point, where some live path needs it 

19 anyway. 

20- Python has one float; ``f32`` computes as f64 here. The WGSL backend is 

21 where narrowing gets real (§2.10 ``type_map``). 

22- Results return natively (a Python value) — the ``ResultPlan`` mirror 

23 becomes physical with DPS targets (ch10); ch08 already proved it round-trips. 

24 

25Composite (tuple-typed) *arguments* are refused loudly for now: their leaf 

26slots exist in the plan, but the IR keeps one logical ``core.param`` — the 

27arg-side normalize lands with the arrays step (see the step-7 review note). 

28"""

40_BIN = {"core.add": "+", "core.sub": "-", "core.mul": "*", "core.div": "/", "core.mod": "%", "core.pow": "**"} 

41_PREDS = {"lt": "<", "gt": ">", "le": "<=", "ge": ">=", "eq": "==", "ne": "!="} 

42_CASTS = {"f64": "float", "f32": "float", "i64": "int", "i32": "int", "u64": "int", "u32": "int", "bool": "bool"}

45def render(region: Region, plan: PackPlan, backend=None, name: str = "kernel") -> str: 

46 """Legalized Region -> Python source. One assignment per node, DAG-shared 

47 nodes emitted once in their OWNER region (the shared dominator-placed 

48 walker in ``_emit`` — the lazy-branch rule), names dense in topo order.""" 

49 args_by_index = {s.source.index: s for s in plan.slots if s.source.root == "arg" and not s.source.sub} 

50 

51 def expr_of(node: Node, names: dict) -> str: 

52 attrs = dict(node.attrs) 

53 arg = [names[id(a)] for a in node.args] 

54 if node.op == "core.param": 

55 spec = args_by_index.get(attrs["index"]) 

56 if spec is None: 

57 raise VerifyError(f"composite argument {attrs['index']} has no scalar slot (arrays step)") 

58 return f"_u({spec.dest.fmt!r}, staging, {spec.dest.offset})[0]" 

59 if node.op == "abi.slot": 

60 return f"_u({attrs['fmt']!r}, staging, {attrs['offset']})[0]" 

61 if node.op == "array.buffer": # leaves-channel position, resolved from THE plan (like abi offsets) 

62 chan = [s for s in plan.slots if s.dest is None] 

63 for k, s in enumerate(chan): 

64 if s.source.root == "env" and (s.source.index, *s.source.sub) == (*attrs["slot"], 0): 

65 return f"leaves[{k}]" 

66 raise VerifyError(f"no buffer leaf for capture path {attrs['slot']!r}") 

67 if node.op == "core.const": 

68 v = attrs["value"] 

69 if isinstance(v, float) and not math.isfinite(v): 

70 return f"float({str(v)!r})" # repr(inf) is not a Python literal 

71 return repr(v) 

72 if node.op in _BIN: 

73 # Numeric policy (070): TRUNC div/mod, like C — never Python's floored 

74 # // %. Integers go through EXACT helpers (float rounding would lose 

75 # precision past 2^53); float mod is math.fmod (sign of the dividend). 

76 if node.op in ("core.div", "core.mod") and isinstance(node.type, Scalar) and node.type.kind[0] in "iu": 

77 return f"{'_tdiv' if node.op == 'core.div' else '_tmod'}({arg[0]}, {arg[1]})" 

78 if node.op == "core.mod": 

79 return f"math.fmod({arg[0]}, {arg[1]})" 

80 return f"{arg[0]} {_BIN[node.op]} {arg[1]}" 

81 if node.op == "core.neg": 

82 return f"-{arg[0]}" 

83 if node.op == "core.cmp": 

84 return f"{arg[0]} {_PREDS[attrs['pred']]} {arg[1]}" 

85 if node.op == "core.cast": 

86 to = attrs["to"] 

87 if not isinstance(to, Scalar): 

88 raise VerifyError(f"python backend cannot cast to {to!r}") 

89 return f"{_CASTS[to.kind]}({arg[0]})" 

90 if node.op == "core.select": 

91 return f"({arg[1]} if {arg[0]} else {arg[2]})" 

92 if node.op in ("core.vec", "core.tuple"): 

93 return "(" + ", ".join(arg) + ("," if len(arg) == 1 else "") + ")" 

94 if node.op == "core.extract": 

95 return f"{arg[0]}[{attrs['index']}]" 

96 table = backend.code_for_op if backend is not None else CODE_FOR_OP 

97 template = table.get(node.op) # surface D: THIS record's spellings (extend()-local) 

98 if template: 

99 return template.format(*arg) 

100 if template is None and node.op in table: # spell(None) claims native support this renderer lacks 

101 raise VerifyError(f"{node.op!r} was spelled None ('native') but this renderer has no native handling") 

102 raise VerifyError(f"python backend has no rendering for {node.op!r}") 

103 

104 def statement(node, nm): 

105 return f"{nm[id(node)]} = {expr_of(node, nm)}" 

106 

107 def branch_join(node, nm, result_of, emit_block, path, ind): 

108 res = nm[id(node)] 

109 out = [f"{ind}if {nm[id(node.args[0])]}:"] 

110 out += emit_block((*path, (id(node), 0)), ind + " ") 

111 out.append(f"{ind} {res} = {result_of(0)}") 

112 out.append(f"{ind}else:") 

113 out += emit_block((*path, (id(node), 1)), ind + " ") 

114 out.append(f"{ind} {res} = {result_of(1)}") 

115 return out 

116 

117 def loop_join(node, nm, result_of, emit_block, path, ind): 

118 res, (lo, hi, init) = nm[id(node)], (nm[id(a)] for a in node.args) 

119 iv, carry = node.regions[0].params 

120 out = [f"{ind}{res} = {init}"] # res IS the carry between iterations 

121 out.append(f"{ind}for {nm[id(iv)]} in range({lo}, {hi}):") 

122 out.append(f"{ind} {nm[id(carry)]} = {res}") 

123 out += emit_block((*path, (id(node), 0)), ind + " ") 

124 out.append(f"{ind} {res} = {result_of(0)}") 

125 return out 

126 

127 lines, names, result = emit_dominated(region, statement, branch_join, indent=" ", loop=loop_join) 

128 body = "\n".join(lines) or " pass" 

129 head = ( 

130 "import math\nfrom struct import unpack_from as _u\n" 

131 "from pdum.dsl.kernel.registry import Out as _Out\n\n" 

132 "def _tdiv(a, b): # exact trunc division (numeric policy: C semantics)\n" 

133 " q = a // b\n" 

134 " return q + 1 if q < 0 and q * b != a else q\n\n" 

135 "def _tmod(a, b):\n" 

136 " return a - _tdiv(a, b) * b\n\n" 

137 f"def {name}(staging, leaves):\n" 

138 ) 

139 guard = ( # buffer leaves are welcome (arrays, ch12); launcher data is peeled BY TYPE (§ Out contract) 

140 " if any(isinstance(x, _Out) for x in leaves):\n" 

141 ' raise TypeError("the python backend takes no launcher data (out= is for device targets)")\n' 

142 ) 

143 return f"{head}{guard}{body}\n return {names[id(result)]}\n"

146def compile_source(source: str, name: str = "kernel"): 

147 ns: dict = {} 

148 exec(compile(source, f"<pdum-python:{name}>", "exec"), ns) # noqa: S102 — this IS the backend 

149 artifact = ns[name] 

150 artifact.__pdum_source__ = source # the artifact carries its own listing (ch09 autopsy) 

151 return artifact

154CODE_FOR_OP: dict = { 

155 "core.tuple": None, # native tuple spelling; batteries add math.* templates 

156 "array.load": "{0}[{1}]", # static, like the C target's — no install-order dependence 

157}

168def install(registry) -> None: 

169 """The explicit seam (same shape as ``stdlib.install``). Each registry 

170 gets its OWN record copy with a fresh spelling table — module singletons 

171 shared across registries would let one registry's spell() rewrite 

172 another's world (review-caught)."""
